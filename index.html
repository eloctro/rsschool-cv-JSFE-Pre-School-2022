          function isPalindrome(line) {
            line = String(line);
            let start = 0;
            let finish = line.length - 1;
            while (finish >= start) {
              if (line[start] !== line[finish]) {
                return false;
              };
              start++;
              finish--;
            };
            return true;
          };
